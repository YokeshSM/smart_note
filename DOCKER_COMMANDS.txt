# Smart Note – Docker helper commands

# Set by release workflow on tag push (e.g. v1.2.3). Edit manually for local builds.
VERSION="v0.0.0"
IMAGE="yokeshsm783/smart-note:${VERSION}"
LOCAL_IMAGE="smart-note:${VERSION}"

## Build (with your Supabase project)

# Replace the values after = with YOUR Supabase URL + anon key.
# Run from the repo root.
sudo docker build \
  --build-arg VITE_SUPABASE_URL="https://YOUR-PROJECT.supabase.co" \
  --build-arg VITE_SUPABASE_ANON_KEY="YOUR_ANON_KEY" \
  -t "${LOCAL_IMAGE}" .

## Run locally

# Expose on localhost:4173 (auto restart unless stopped)
sudo docker run --restart=always -p 4173:80 "${LOCAL_IMAGE}"
sudo docker run --restart=always -p 4173:80 yokeshsm783/smart-note:latest


## Push image to Docker Hub (or other registry)

# Log in first if needed: docker login
sudo docker tag "${LOCAL_IMAGE}" "${IMAGE}"
sudo docker push "${IMAGE}"

## Pull image (for other users)

sudo docker pull "${IMAGE}"

## Run pulled image

sudo docker run --restart=always -p 4173:80 "${IMAGE}"
